// Lightbox Functionality
                document.addEventListener("DOMContentLoaded", function () {
                  // Create Lightbox Overlay
                  const lightboxOverlay = document.createElement("div");
                  lightboxOverlay.className = "lightbox-overlay";
                  lightboxOverlay.innerHTML = `
                    <img />
                    <span class="lightbox-close">&times;</span>
                  `;
                  document.body.appendChild(lightboxOverlay);
                
                  // Open Lightbox on Image Click
                  document.addEventListener("click", (e) => {
                    if (e.target.classList.contains("lightbox-image")) {
                      const imgSrc = e.target.getAttribute("data-fullsize");
                      lightboxOverlay.querySelector("img").src = imgSrc;
                      lightboxOverlay.style.display = "flex";
                    }
                    // Close Lightbox
                    if (e.target.classList.contains("lightbox-close") || e.target === lightboxOverlay) {
                      lightboxOverlay.style.display = "none";
                    }
                  });
                });
